    // ========================================
    // Complaints Collection (CSV Export)
    // ========================================
    
    match /complaints/{complaintId} {
      // Helper function to check complaint ownership
      function isComplaintOwner() {
        return isAuthenticated() &&
               (resource.data.userId == request.auth.uid ||
                resource.data.user_id == request.auth.uid);
      }
      
      // âœ… Allow authenticated users to list complaints
      allow list: if isAuthenticated();
      
      // Read individual complaint (user must own it or be admin)
      allow get: if isComplaintOwner() || isAdmin();
      
      // Create complaint (user must set themselves as the complainant)
      allow create: if isAuthenticated() &&
                       (request.resource.data.userId == request.auth.uid ||
                        request.resource.data.user_id == request.auth.uid);
      
      // Users can update their own pending complaints (before admin responds)
      // Admins can update any complaint
      allow update: if isAdmin() ||
                       (isComplaintOwner() &&
                        resource.data.status == 'pending');
      
      // Only admin can delete complaints
      allow delete: if isAdmin();
    }

âœ… ALLOW LIST: Any authenticated user can query complaints
   - But individual documents still require ownership check

âœ… ALLOW GET: User can read their own complaints OR admin can read all
   - Checks if complaint's user_id matches current user's auth.uid

âœ… ALLOW CREATE: User can submit complaint, must be the complainant
   - Validates request.resource.data.user_id == request.auth.uid
   - Supports both userId and user_id field names

âœ… ALLOW UPDATE: Admin can update any, users can update pending only
   - Users can only update if status is 'pending'
   - Once admin responds, user cannot modify

âœ… ALLOW DELETE: Only admin can delete complaints
   - Users cannot delete their own complaints
