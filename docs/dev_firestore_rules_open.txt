// Purpose: Allow unrestricted access during app development

rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    
    // ===================================================================
    // ALLOW ALL ACCESS TO ALL COLLECTIONS
    // ===================================================================
    // Anyone can read and write any document in any collection
    
    match /{document=**} {
      allow read, write: if true;
    }
  }
}

// WHAT THIS MEANS
// ===================================================================
//
// ✅ ALL users can read ANY document
// ✅ ALL users can write/update/delete ANY document
// ✅ NO authentication required
// ✅ NO permission checks
// ✅ NO field validation
// ✅ NO restrictions whatsoever
//
// BENEFITS FOR DEVELOPMENT:
// ✅ No "permission-denied" errors
