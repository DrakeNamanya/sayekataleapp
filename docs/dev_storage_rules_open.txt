// ===================================================================
// FIREBASE STORAGE SECURITY RULES - DEVELOPMENT MODE (OPEN)
// ===================================================================
// ⚠️ WARNING: These rules allow FULL ACCESS to all files
// ⚠️ USE ONLY FOR DEVELOPMENT - NOT FOR PRODUCTION!
// ===================================================================
// Apply these in Firebase Console → Storage → Rules tab
// Purpose: Allow unrestricted file uploads/downloads during development
// ===================================================================

rules_version = '2';

service firebase.storage {
  match /b/{bucket}/o {
    
    // ===================================================================
    // ALLOW ALL ACCESS TO ALL PATHS
    // ===================================================================
    // Anyone can read and write any file at any path
    
    match /{allPaths=**} {
      allow read, write: if true;
    }
  }
}

// ===================================================================
// WHAT THIS MEANS
// ===================================================================
//
// ✅ ALL users can upload files to ANY path
// ✅ ALL users can download ANY file
// ✅ ALL users can delete ANY file
// ✅ NO authentication required
// ✅ NO file size limits
// ✅ NO content type restrictions
// ✅ NO path restrictions
// ✅ NO rate limiting
//
// BENEFITS FOR DEVELOPMENT:
// ✅ No "firebase image timeout" errors
// ✅ No "permission denied" errors
// ✅ Upload/download files instantly
// ✅ Test any storage feature freely
// ✅ No IAM permission prompts
// ✅ Focus on app functionality
//
// ⚠️ IMPORTANT REMINDERS:
// ❌ DO NOT use these rules in production
// ❌ Anyone can upload/delete files
// ❌ No protection against abuse (large files, spam)
// ❌ Storage costs could increase rapidly
// ❌ Change to secure rules before launching
//
// ===================================================================
